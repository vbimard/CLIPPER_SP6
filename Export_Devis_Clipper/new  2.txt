SELECT ARTICLEM.AMCLEUNIK,DIMENSIO.DMCLEUNIK,ARTICLEM.COARTI,DIMENSIO.DIM1 as LNG,DIMENSIO.DIM2 AS LARGEUR,MULTISIT.QTE, DIMENSIO.CHUTE, ARTICLEM.DIM1 as EP,
Tech_NuanceMatiere.Nuance AS CODENUANCE,Tech_EtatMatiere.Etat AS CODEETAT,Tech_NuanceMatiere.Densite,
FAMILLE.DIMENSIONS, FAMILLE.MULTIDIM,
DIMENSIO.AMCLEUNIK AS DIMENSIO_AMCLEUNIK,
ARTICLEM.DATESO,ARTICLEM.ETATMAT, ARTICLEM.NUANCE, ARTICLEM.PRIXART, ARTICLEM.PMP, ARTICLEM.DERNPRIX, ARTICLEM.LIBRETXT4,ARTICLEM.COFA, ARTICLEM.DIM2 , ARTICLEM.DIM3
FROM ARTICLEM  
LEFT JOIN FAMILLE  ON FAMILLE.COFA=ARTICLEM.COFA 
LEFT JOIN Tech_EtatMatiere  ON Tech_EtatMatiere.Libelle=ARTICLEM.ETATMAT
LEFT JOIN Tech_NuanceMatiere  ON Tech_NuanceMatiere.Libelle=ARTICLEM.NUANCE 
LEFT JOIN DIMENSIO  ON DIMENSIO.	AMCLEUNIK=ARTICLEM.AMCLEUNIK
LEFT JOIN MULTISIT  ON MULTISIT.DMCLEUNIK=DIMENSIO.DMCLEUNIK
WHERE  ARTICLEM.CHUTE<>'O' AND FAMILLE.DIMENSIONS=3 AND FAMILLE.MULTIDIM=1 AND ARTICLEM.DATESO='' AND DIMENSIO.CHUTE='1';
